<script>
    function goToStep(step) {
        document.querySelectorAll('.step-content').forEach(c => c.classList.remove('active'));
        document.getElementById(`step-${step}`).classList.add('active');
        document.querySelectorAll('.step-dot').forEach((dot, i) => {
            dot.className = (i + 1 < step) ? 'step-dot completed' : (i + 1 === step) ? 'step-dot active' : 'step-dot';
        });
        document.getElementById('page-title').innerText = step === 4 ? "Investment Opportunities" : "Investment Growth Analysis";
    }

    function processStep2() {
        const init = parseFloat(document.getElementById('init-amount').value) || 0;
        const curr = parseFloat(document.getElementById('curr-amount').value) || 0;
        const profit = curr - init;
        const roi = init !== 0 ? (profit / init) * 100 : 0;

        document.getElementById('stat-profit').innerText = '€' + profit.toLocaleString();
        document.getElementById('stat-roi').innerText = roi.toFixed(1) + '%';
        document.getElementById('stat-roi').style.color = profit >= 0 ? 'var(--success)' : 'var(--danger)';
        goToStep(3);
    }
</script>
